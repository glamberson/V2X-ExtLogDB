

Public Sub InsertMRLLineItems(batchData As String)
    Dim conn As ADODB.Connection
    Dim cmd As ADODB.Command
    
    Set conn = GetPostgreSQLConnection()
    Set cmd = New ADODB.Command
    
    With cmd
        .ActiveConnection = conn
        .CommandType = adCmdText
        .CommandText = "CALL insert_mrl_line_items(?)"
        .Parameters.Append .CreateParameter("@batch_data", adLongVarChar, adParamInput, , batchData)
        .Execute
    End With
    
    conn.Close
End Sub
