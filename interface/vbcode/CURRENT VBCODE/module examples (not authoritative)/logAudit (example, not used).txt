Public Function LogAudit(action As String, orderLineItemId As Long, fulfillmentItemId As Long, details As String, updateSource As String)
    Dim cmd As Object
    Set cmd = CreateObject("ADODB.Command")
    cmd.ActiveConnection = g_conn
    cmd.CommandText = "SELECT log_audit(?, ?, ?, ?, ?, ?)"
    cmd.Parameters.Append cmd.CreateParameter("@action", 200, 1, 255, action)
    cmd.Parameters.Append cmd.CreateParameter("@order_line_item_id", 3, 1, , orderLineItemId)
    cmd.Parameters.Append cmd.CreateParameter("@fulfillment_item_id", 3, 1, , fulfillmentItemId)
    cmd.Parameters.Append cmd.CreateParameter("@changed_by", 3, 1, , userId)
    cmd.Parameters.Append cmd.CreateParameter("@details", 200, 1, 255, details)
    cmd.Parameters.Append cmd.CreateParameter("@update_source", 200, 1, 255, updateSource)
    cmd.Execute
    Set cmd = Nothing
End Function
