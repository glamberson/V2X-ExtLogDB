
' Initialize Postgres Connection

Public Sub InitializePostgresConnection()
    If g_conn Is Nothing Then
        g_connString = "Driver={PostgreSQL Unicode};" & _
                   "Server=localhost;" & _
                   "Port=5432;" & _
                   "Database=Beta_003;" & _
                   "Uid=login;" & _
                   "Pwd=FOTS-Egypt;" & _
                   "TextSizeMax=2147483647;" & _
                   "LongVarcharSize=2147483647;"
    End If
End Sub
