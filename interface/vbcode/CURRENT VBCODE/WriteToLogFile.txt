
' WRITE TO LOG FILE
' version 0.9.44

Public Sub WriteToLogFile(filePath As String, message As String)
    Dim fileNum As Integer
    fileNum = FreeFile
    
    Open filePath For Append As #fileNum
    Print #fileNum, Format(Now, "yyyy-mm-dd hh:nn:ss") & " - " & message
    Close #fileNum
End Sub
