Sub ImportExcelToAccess(xlSheet As Object)
    Dim db As DAO.Database
    
    ' Import the Excel sheet data into the Access TempTable
    ' This assumes that the sheet has been cleaned and validated before calling this function.
    DoCmd.TransferSpreadsheet acImport, acSpreadsheetTypeExcel12, "TempTable", xlSheet.Parent.FullName, True, xlSheet.Name & "!"
    
    Set db = Nothing
End Sub
